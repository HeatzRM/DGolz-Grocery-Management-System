CREATE DATABASE DGOLZMART
USE DGOLZMART


CREATE TABLE Employee
(
EmpID int identity(1001,1) primary key,
Lname varchar(50),
Fname varchar(50),
Mi varchar(50),
ContactNo varchar(50),
EmpAddress varchar(100)
)


CREATE TABLE Product
(
ProdID int identity(1001,1) primary key,
ProductName varchar(50),
Unit float,
UnitPrice float,
ItemDesc varchar(50),
Quantity varchar(50),
Manufacturer varchar(50),
ManuDate Datetime,
ExpDate Datetime
)

CREATE TABLE BuyingTransaction
(
BuyingTransID int identity(1001,1) primary key,
Quantity float,
TransDate datetime,
Tax float,
EmpID int,
CustID int,
ProdID int
)


CREATE TABLE SupplyTransaction
(
SupplyTransID int IDENTITY (101,1) PRIMARY KEY,
OrderDate Datetime,
OrderDesc varchar (50),
Quantity int,
DeliveryDate Datetime,
ProductProdID int,
SupplierSupplierID int

FOREIGN KEY(ProductProdID)REFERENCES Product(ProdID),
FOREIGN KEY(SupplierSupplierID)REFERENCES Supplier(SupplierID)

)

CREATE TABLE Receipt
(
ReceiptID int IDENTITY (101,1) PRIMARY KEY,
ReceiptDateTime datetime,
RegisterNumber int,
BuyingTransactionBuyingTransID int


FOREIGN KEY(BuyingTransactionBuyingTransID)REFERENCES BuyingTransaction(BuyingTransID)
)
